\begin{PROBLEM}
    تعداد رشته‌های به طول 
    $2n$
    متشکل از
    $n$
    حرف 
  $A$
    و
    $n$
    حرف
  $B$
    را بیابید که اگر از سمت چپ شروع به حرکت کنیم در هیچ نقطه ای, تعداد 
  $B$
   ‌های سمت چپ از تعداد 
  $A$
  ‌ها بیشتر نشود.
   \SOLUTION{
    تعداد این رشته‌ها با تعداد پرانتز‌های مجاز معادل است, به این صورت که حرف
    $A$
    معادل پرانتز باز و حرف
    $B$
    معادل پرانتز بسته است.

   }
  \end{PROBLEM}
