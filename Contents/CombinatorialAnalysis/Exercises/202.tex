\EXERCISE
\begin{enumerate}
\item
تعداد توابع یک به یک از
$A = \{a_1, a_2, a_3, \cdots, a_n\}$
به
$B = \{b_1, b_2, b_3, \cdots, b_m\}$
را به ازای تمام حالات ممکن
$m$
و
$n$
محاسبه کنید.
\item
تعداد توابع پوشا از
$A = \{a_1, a_2, a_3, \cdots, a_n\}$
به
$B = \{b_1, b_2, b_3, \cdots, b_m\}$
را به ازای تمام حالات ممکن
$m$
و
$n$
محاسبه کنید.
\item
از قسمت ب) یک رابطه بر حسب
$m$
و
$n$
به‌دست می‌آید. اگر
$\frac{1}{n!}$
برابر آن را
$s(m, n)$
بنامیم، نشان دهید:
$$s(m + 1, n) = s(m, n - 1) + ns(m, n)$$
\end{enumerate}
