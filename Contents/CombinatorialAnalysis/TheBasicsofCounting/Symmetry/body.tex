\SUBSECTION{اصل تقسیم (تقارن)}

\begin{DEFINITION}
    \p
    \FOCUSEDON{اصل تقسیم:}
    اگر بتوان تعداد 
    $N$
    حالت انجام یک فرایند را شمرد که بین هر
    $k$
    حالت تقارن وجود داشته باشد
    (نامتمایز باشند)،
    تعداد حالات متمایز شمرده شده برای انجام فرایند، برابر است با:
    $$N/k$$
\end{DEFINITION}
